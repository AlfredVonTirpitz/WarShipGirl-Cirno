country_event = {
	id = Cirno_sex_event.1
	title = Cirno_sex_event.1.title
	desc = Cirno_sex_event.1.desc
	picture = Cirno_sex_eventPicture1
	goto = from
	is_triggered_only = yes
	trigger = {
	  ai = no
		#NOT = { has_country_flag = Cirno_sex_event_flag1 } #仅触发一次
		FROM = {
			fort_level = 1
			development = 20
			owner = { NOT = { tag = ROOT } }
			OR = { #五十年一次
				NOT = { has_country_flag = Cirno_sex_event_flag1 }
				had_country_flag = { flag = Cirno_sex_event_flag1 days = 18250 }
			}
			OR = { #被围攻省份五十年内不能再次触发
				NOT = { has_province_flag = Cirno_SexSiegeWon_province_flag }
				had_province_flag = { flag = Cirno_SexSiegeWon_province_flag days = 18250 }
			}
		}
	}
	immediate = {
		hidden_effect = {
			set_country_flag = Cirno_sex_event_flag1
			FROM = { set_province_flag = Cirno_SexSiegeWon_province_flag }
		}
	}
	#mean_time_to_happen = { days = 1 }
	option = {
		name = Cirno_sex_event.1.a
		add_dip_power = 6
		add_sailors = 1
	}
	option = {
		name = Cirno_sex_event.1.b
		add_mil_power = 6
		add_sailors = 1
	}
	option = {
		name = Cirno_sex_event.1.c
		add_mil_power = 8
	}
}
